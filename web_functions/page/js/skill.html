    alert("button-click");
    // 阻止事件冒泡到DOM树上
    event.stopPropagation(); // 只执行button的click，如果注释掉该行，将执行button、p和div的click （同类型的事件）  
} );

深拷贝

function deepClone(obj) {
    if (typeof obj !== 'object') return obj;
    if (JSON) {
        let str = JSON.stringify(obj);
        return JSON.parse(str);
    }
}
